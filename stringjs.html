W JavaScript tak samo jak w innych językach możemy operować nie tylko na liczbach, ale także na tekstach. 
const text = "Ala ma kota.";
const age = 21;
const text = "Ten kot ma " + age + " lat";
const text = 'Ten pies ma ' + age + ' lat';
//tak wstawiamy zmienną do tekstu
